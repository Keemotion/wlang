      def self.downcase(parser, offset)
        expression, reached = parser.parse(offset, "wlang/ruby")
        value = parser.evaluate(expression)
        value = value.nil? ? "" : value.to_s
        result = EncoderSet::PlainText.downcase(value)
        [result, reached]
      end
