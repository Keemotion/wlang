      def column_of(str, index)
        return 1 if index == 0
        newline_index = str.rindex("\n", index - 1)
        if newline_index
          index - newline_index
        else
          index + 1
        end
      end

      def indent(str, n)
        if n >= 0
          str.gsub(/^/, ' ' * n)
        else
          str.gsub(/^ {0,#{-n}}/, "")
        end
      end

strip_block(str)

Strips a multiline block.

Example: 

  ([\t ]*\n)?
  ([\t ]\n)*
  [\t ]*some text here\n
  [\t ]*  indented also\n?
  [\t ]*

becomes: 

  some text here\n
    indented also\n

[Source]

# File lib/wlang/intelligent_buffer.rb, line 48
      def strip_block(str)
        match = str.match(/\A[\t ]*\n?/)
        str = match.post_match if match
        match = str.match(/\n[\t ]*\Z/)
        str = (match.pre_match << "\n") if match
        str
      end

      def tabto(str, n)
        if str =~ /^( *)\S/
          indent(str, n - $1.length)
        else
          str
        end
      end
